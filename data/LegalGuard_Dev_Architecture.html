Database Schema — 6 Tables หลัก
CREATE TABLE judgments ( id UUID PRIMARY KEY, case_no VARCHAR(50) NOT NULL, court_type ENUM('supreme','appeal', 'district','admin'), year INT, province VARCHAR(100), statute_refs JSON, content TEXT, -- anonymized embedding VECTOR(1024), -- pgvector cfs_score FLOAT, pii_masked BOOLEAN DEFAULT TRUE );
CREATE TABLE audit_logs ( id UUID PRIMARY KEY, query_hash VARCHAR(64), -- SHA-256 user_id VARCHAR(100), h_score FLOAT, action VARCHAR(50), timestamp TIMESTAMP, prev_hash VARCHAR(64) -- chain! ); -- IMMUTABLE — no DELETE allowed
CREATE TABLE metadata ( id UUID PRIMARY KEY, judgment_id UUID REFERENCES judgments, statute_no VARCHAR[], citations UUID[], keywords TEXT[], summary TEXT );
CREATE TABLE pii_masks ( id UUID PRIMARY KEY, doc_id UUID, spans JSON, -- [{start,end,type}] masked_text TEXT, created_at TIMESTAMP );
CREATE TABLE vector_index_mapping ( id UUID, judgment_id UUID, vector_id BIGINT, -- FAISS ID chunk_idx INT, embedding_hash VARCHAR(64) );
FAISS Index Configuration
# FAISS IVF+PQ Configuration d = 1024 # WangchanX embedding dimension nlist = 1000 # IVF clusters (√127800 ≈ 357, ×3 = 1000) M = 64 # PQ subvectors nbits = 8 # bits per subvector index = faiss.IndexIVFPQ(quantizer, d, nlist, M, nbits) index.nprobe = 10 # search 10 clusters (speed/recall tradeoff) # Index stats Total vectors: 127,800 Index size: 2.1 GB (vs 512 MB raw — 512x compression) Query time: <50ms avg
Data Ingestion Pipeline — Document → Vector Index
PDF / TXT / Hugging Face Datasets (API) → Federated Ingestion Engine (mjs) → PII Masking → Text Chunking (512 tok, 50 overlap) → WangchanX Embed (1024-dim) → FAISS IVF/PQ Index → PostgreSQL + vector_index_mapping

Code: Hybrid Search Implementation

# Layer 1 — Hybrid Retrieval (Python) import faiss, numpy as np from rank_bm25 import BM25Okapi # Step 1: FAISS semantic search (70%) query_vector = wangchanx_model.encode(query_text) query_vector = query_vector / np.linalg.norm(query_vector) D, I = faiss_index.search( query_vector.reshape(1, -1), k=50 ) faiss_scores = normalize(1 - D[0] / D[0].max()) # Step 2: BM25 keyword search (30%) tokenized = pythainlp_tokenize(query_text) bm25_raw = bm25_index.get_scores(tokenized) bm25_scores = normalize(bm25_raw) # Step 3: Hybrid combine hybrid_scores = 0.7 * faiss_scores + 0.3 * bm25_scores # Step 4: Final reranking score final = 0.5*similarity + 0.3*cfs + 0.2*confidence top3 = final.argsort()[-3:][::-1]

Code: CBB Safety Gate

# Layer 5 — CBB Confidence Cap def compute_p_responsible(posterior, uu_penalty, u_ethical=0.98, margin=0.02): option_a = u_ethical * uu_penalty option_b = posterior + margin p_resp = min(option_a, option_b) if p_resp < 0.90: return "ABSTAIN", p_resp # ← trigger abstention return "RESPOND", p_resp # PII Masking Pipeline def mask_pii(text): entities = ner_model.predict(text) # NER patterns = regex_scanner.scan(text) # Pattern match rules = rule_engine.apply(text) # Rules all_spans = merge_spans(entities + patterns + rules) return replace_spans(text, all_spans) # Recall 99.2% # CAL-130 Audit Chain def log_query(query_hash, prev_hash, h_score): record = {"hash": sha256(query_hash + prev_hash), ...} db.insert("audit_logs", record) # NEVER delete
